import csv
import argparse
import datetime


class Supermarket:
    def __init__(self, product_name, count, buy_price, expiration_date):
        self.product_name = product_name
        self.count = count
        self.buy_price = buy_price
        self.expiration_date = expiration_date
        self.bought_id = None
        self.sell_date = None
        self.sell_price = None

    def buy(self, bought_id):
        self.bought_id = bought_id

    def sell(self, bought_id, count, sell_price):
        if self.bought_id == bought_id:
            self.count -= count
            self.sell_date = datetime.date.today()
            self.sell_price = sell_price


# Update the buy_product function to reflect the new structure and to update both bought.csv and inventory.csv


def buy_product_v2(product, name, buy_price, sell_price, expiration_date, quantity):
    date_now = datetime.datetime.now().strftime("%Y-%m-%d")
    time_now = datetime.datetime.now().strftime("%H:%M:%S")

    # Calculate the report_price (for now, let's take the average of buy_price and sell_price)
    report_price = (float(buy_price) + float(sell_price)) / 2

    # Get the next available ID for bought.csv
    with open("bought.csv", "r") as csvfile:
        reader = csv.reader(csvfile)
        id_value = sum(1 for _ in reader)

    # Append the product details to bought.csv
    with open("bought.csv", "a", newline="") as csvfile:
        fieldnames = [
            "id",
            "product",
            "name",
            "buy_price",
            "sell_price",
            "date",
            "time",
            "report_price",
            "expiration_date",
            "quantity",
        ]
        writer = csv.DictWriter(csvfile, fieldnames=fieldnames)

        writer.writerow(
            {
                "id": id_value,
                "product": product,
                "name": name,
                "buy_price": buy_price,
                "sell_price": sell_price,
                "date": date_now,
                "time": time_now,
                "report_price": report_price,
                "expiration_date": expiration_date,
                "quantity": quantity,
            }
        )

    # Append the product details to inventory.csv
    with open("path_to_your_directory/inventory.csv", "a", newline="") as csvfile:
        fieldnames = [
            "id",
            "product",
            "name",
            "buy_price",
            "sell_price",
            "date",
            "time",
            "report_price",
            "expiration_date",
            "quantity",
        ]
        writer = csv.DictWriter(csvfile, fieldnames=fieldnames)

        writer.writerow(
            {
                "id": id_value,
                "product": product,
                "name": name,
                "buy_price": buy_price,
                "sell_price": sell_price,
                "date": date_now,
                "time": time_now,
                "report_price": report_price,
                "expiration_date": expiration_date,
                "quantity": quantity,
            }
        )

    return f"Added {quantity} {name} ({product}) to inventory."

# Update the report_inventory function to reflect the new structure and to fetch data from inventory.csv


def report_inventory_v2(date=None):
    # If no date is provided, use the current date
    if date is None:
        date = datetime.datetime.now().strftime("%Y-%m-%d")

    # Read the inventory.csv and fetch product details
    with open("path_to_your_directory/inventory.csv", "r") as csvfile:
        reader = csv.DictReader(csvfile)
        products = [
            row for row in reader if row["date"] <= date
        ]  # Include only products bought on or before the given date

    # Display the products in a tabular format
    table_header = ["Product Name", "Count", "Buy Price", "Expiration Date"]
    table_data = []

    # Prepare the data for display
    for product in products:
        table_data.append(
            [
                product["name"],
                product["quantity"],
                product["buy_price"],
                product["expiration_date"],
            ]
        )

    # Format the table for display
    table_string = "+--------------+-------+-----------+-----------------+\n"
    table_string += "| " + " | ".join(table_header) + " |\n"
    table_string += "+--------------+-------+-----------+-----------------+\n"
    for row in table_data:
        table_string += "| " + " | ".join(map(str, row)) + " |\n"
    table_string += "+--------------+-------+-----------+-----------------+"

    return table_string

def sell_product_corrected(product_name, sell_price, date=None):
    """
    Sells a product and updates the inventory and sold.csv files.

    Args:
    - product_name (str): The name of the product to sell.
    - sell_price (float): The price at which the product is sold.
    - date (str, optional): The date of the sale. Defaults to today's date.

    Returns:
    - str: A message indicating the success or failure of the sale.
    """

    # If no date is provided, use the current date
    if date is None:
        date = datetime.datetime.now().strftime("%Y-%m-%d")

    # Load the inventory
    with open("path_to_your_directory/inventory.csv", "r") as csvfile:
        reader = csv.DictReader(csvfile)
        inventory = [row for row in reader]

    # Check if the product is available in the inventory
    for product in inventory:
        if (
            product[" product_name"].strip() == product_name
            and int(product[" quantity"].strip()) > 0
        ):
            # Reduce the product quantity in the inventory
            product[" quantity"] = str(int(product[" quantity"].strip()) - 1)

            # Update the inventory.csv
            fieldnames = [
                " product_name",
                " quantity",
                " buy_price",
                " expiration_date",
            ]
            with open(
                "path_to_your_directory/inventory.csv", "w", newline=""
            ) as csvfile:
                writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
                writer.writeheader()
                for row in inventory:
                    writer.writerow(row)

            # Update the sold.csv with the sale details
            with open("path_to_your_directory/sold.csv", "a", newline="") as csvfile:
                writer = csv.DictWriter(
                    csvfile, fieldnames=["product_name", "sell_price", "sell_date"]
                )
                writer.writerow(
                    {
                        "product_name": product_name,
                        "sell_price": sell_price,
                        "sell_date": date,
                    }
                )

            return f"Sold {product_name} for {sell_price} on {date}"

    # If product not found or quantity is zero
    return f"ERROR: {product_name} not in stock or not enough quantity."
