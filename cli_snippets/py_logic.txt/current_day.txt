# import

import datetime

# create a file 'current_date_time.txt' to store the current date and time
with open("current_date_time.txt", "w") as f:
    f.write(str(datetime.datetime.now()))


# function to advance the current date and time
def advance_date_time():
    with open("current_date_time.txt", "r") as f:
        current_date_time = datetime.datetime.strptime(
            f.read().strip(), "%Y-%m-%d %H:%M:%S.%f"
        )
    with open("current_date_time.txt", "w") as f:
        f.write(str(current_date_time + datetime.timedelta(days=1)))


# function to get the current date and time
def get_current_date_time():
    with open("current_date_time.txt", "r") as f:
        current_date_time = datetime.datetime.strptime(
            f.read().strip(), "%Y-%m-%d %H:%M:%S.%f"
        )
    return current_date_time
